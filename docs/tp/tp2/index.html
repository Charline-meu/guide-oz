    Question 5 # 

    declare
fun {Take Xs N}
   case Xs of H|T then
      if N>0 then H|{Take T N-1}
      else nil end
      [] nil then nil
   end
end

fun {Drop Xs N}
   case Xs of H|T then
      if N>0 then {Drop T N-1}
      else Xs
      end
   [] nil then nil
   end
end

    declare
fun {Prefix L1 L2}
   case L1 of H1|T1 then
        case L2 of H2|T2 then
	        if H1 == H2 then {Prefix T1 T2}
	        else false
	        end
        [] nil then false
        end
    [] nil then true
    end
end

fun {FindString S T}
    IndexMax = {Length T} - {Length S} + 1
    fun {FindStringHelp L1 L2 I}
        if I =< IndexMax then
            if {Prefix L1 L2} then
                I|{FindStringHelp L1 L2.2 I+1}
            else {FindStringHelp L1 L2.2 I+1}
            end
        else nil
        end
    end
in
    {FindStringHelp S T 1}
end

    declare
fun {Applique L F}
   case L of H|T then {F H}|{Applique T F}
   [] nil then nil
   end
end

fun {Lol X} lol(X) end
{Browse {Applique [1 2 3] Lol}}


fun {MakeAdder N}
   fun{$ X}
      X+N
   end
end

Add5 = {MakeAdder 5}
{Browse {Add5 13}}


fun {AddAll L N}
    Add = {MakeAdder N}
in
    {Applique L Add}
end
{Browse {AddAll [1 2 3] 4}}
